11class AnymailError(Exception): 

12 """Base class for exceptions raised by Anymail 

13 

14 Overrides __str__ to provide additional information about 

15 the ESP API call and response. 

16 """ 

17 

18 def __init__(self, *args, **kwargs): 

19 """ 

20 Optional kwargs: 

21 email_message: the original EmailMessage being sent 

22 status_code: HTTP status code of response to ESP send call 

23 backend: the backend instance involved 

24 payload: data arg (*not* json-stringified) for the ESP send call 

25 response: requests.Response from the send call 

26 raised_from: original/wrapped Exception 

27 esp_name: what to call the ESP (read from backend if provided) 

28 """ 

29 self.backend = kwargs.pop('backend', None) 

30 self.email_message = kwargs.pop('email_message', None) 

31 self.payload = kwargs.pop('payload', None) 

32 self.status_code = kwargs.pop('status_code', None) 

33 self.raised_from = kwargs.pop('raised_from', None) 

34 self.esp_name = kwargs.pop('esp_name', 

35 self.backend.esp_name if self.backend else None) 

36 if isinstance(self, HTTPError): 

37 # must leave response in kwargs for HTTPError 

38 self.response = kwargs.get('response', None) 

39 else: 

40 self.response = kwargs.pop('response', None) 

41 super(AnymailError, self).__init__(*args, **kwargs) 

42 

43 def __str__(self): 

44 parts = [ 

45 " ".join([six.text_type(arg) for arg in self.args]), 

46 self.describe_raised_from(), 

47 self.describe_send(), 

48 self.describe_response(), 

49 ] 

50 return "\n".join(filter(None, parts)) 

51 

52 def describe_send(self): 

53 """Return a string describing the ESP send in self.email_message, or None""" 

54 if self.email_message is None: 

55 return None 

56 description = "Sending a message" 

57 try: 

58 description += " to %s" % ','.join(self.email_message.to) 

59 except AttributeError: 

60 pass 

61 try: 

62 description += " from %s" % self.email_message.from_email 

63 except AttributeError: 

64 pass 

65 return description 

66 

67 def describe_response(self): 

68 """Return a formatted string of self.status_code and response, or None""" 

69 if self.status_code is None: 

70 return None 

71 

72 # Decode response.reason to text -- borrowed from requests.Response.raise_for_status: 

73 reason = self.response.reason 

74 if isinstance(reason, six.binary_type): 

75 try: 

76 reason = reason.decode('utf-8') 

77 except UnicodeDecodeError: 

78 reason = reason.decode('iso-8859-1') 

79 

80 description = "%s API response %d (%s)" % (self.esp_name or "ESP", self.status_code, reason) 

81 try: 

82 json_response = self.response.json() 

83 description += ":\n" + json.dumps(json_response, indent=2) 

84 except (AttributeError, KeyError, ValueError): # not JSON = ValueError 

85 try: 

86 description += ": %r" % self.response.text 

87 except AttributeError: 

88 pass 

89 return description 

90 

91 def describe_raised_from(self): 

92 """Return the original exception""" 

93 if self.raised_from is None: 

94 return None 

95 return ''.join(format_exception_only(type(self.raised_from), self.raised_from)).strip()

125 """Exception for Anymail features that the ESP doesn't support. 

126
